这是模块的输出文件，请 require 或 seajs.use 以上 ID 来使用本文件，样式文件请自行 link。

演示文档

seajs.use('alipay/spassword/1.0.0/spassword', function(spassword){

    new spassword("input[name=spwd]").on("complete", function(value){
        var ME = this;
        window.setTimeout(function(){
            ME.clear();
        }, 500);
    }).render();

});
